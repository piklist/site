Here is the code used to create the taxonomy and hide the original metabox

//Create Project Status Taxonomy
add_filter('piklist_taxonomies', 'project_status');
  function project_status($taxonomies) {
     $taxonomies[] = array(
        'post_type' => 'mro_project'
        ,'name' => 'project_status'
        ,'show_admin_column' => true
        ,'configuration' => array(
          'hierarchical' => true
          ,'labels' => piklist('taxonomy_labels', 'Project State')
          ,'hide_meta_box' => true
          ,'show_ui' => true
          ,'query_var' => true
          ,'rewrite' => array(
            'slug' => 'project-status'
          )
        )
      );
    return $taxonomies;
  }
