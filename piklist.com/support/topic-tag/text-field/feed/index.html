add_filter('piklist_post_types', 'grantee_post_type');
 function grantee_post_type($post_types)
 {
  $post_types['grantee'] = array(
    'labels' => piklist('post_type_labels', 'Grants')
    ,'title' => __('Grants')
    ,'public' => true
    ,'rewrite' => array(
      'slug' => 'grantee'
    )
    ,'supports' => array(
      'author'
      ,'revisions'
      ,'title'
      ,'partners'
    )
    ,'hide_meta_box' => array(
      'slug'
      ,'author'
      ,'revisions'
      ,'comments'
      ,'commentstatus'
    )
    ,'edit_columns' => array(
       'title' => __('Grantee')
       ,'author' => __('customer')
       // ,'partners' => __('Partners')
     )
  );
return $post_types;
}
