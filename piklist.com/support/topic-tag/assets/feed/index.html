]]> Hello,
Currently, when I add assets to my plugin, CSS and Scripts are loaded and are available in all pages across the admin backend. How can I go about specifying where to load them. Ex: a custom post type, or a Metabox.

When I load my plugin I have something like:


add_filter('piklist_assets', 'plugin_assets');
function plugin_assets($assets)
{    
    array_push($assets['scripts'], array(
        'handle'    => 'plugin-scripts',
        'src'       => get_template_directory_uri () .'/piklist/parts/js/scripts.js',
        'ver'       => '1.0',
        'deps'      => 'jquery',
        'enqueue'   => true,
        'in_footer' => true,
        'admin'     => true,
        'front'     => false
    ));

    array_push($assets['styles'], array(
        'handle'    => 'plugin-styles',
        'src'       => get_template_directory_uri () . '/piklist/parts/css/plugin-styles.css',
        'ver'       => '1.2',
        'enqueue'   => true,
        'in_footer' => true,
        'admin'     => true,
        'media'     => 'screen, projection',
    ));

    array_push($assets['styles'], array(
        'handle'    => 'plugin-fonts',
        'src'       => get_template_directory_uri () . '/css/font-awesome.css',
        'ver'       => '1.2',
        'enqueue'   => true,
        'in_footer' => false,
        'admin'     => true,
        'media'     => 'screen, projection',
    ));

    return $assets;
}

Thanks
