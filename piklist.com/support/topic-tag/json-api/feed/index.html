]]> Hello piklist friends,

I was wondering if anyone would know how to number the order of the drag and drop fields of an add_more group?

I would like to eventually be able to have citations below a post, which are sortable, and labeled by their order in the editor view.


piklist('field', array(
  'type' => 'group'
  ,'label' => __('Sources')
  ,'description' => __('Any additional sources to cite')
  ,'field' => 'sources_group'
  ,'add_more' => true
  ,'fields' => array(
    array(
      'type' => 'text'
      ,'field' => 'source_title'
      ,'label' => 'Source Title'
      ,'columns' => 12
      ,'attributes' => array(
        'class' => 'large-text'
      )
    )
    ,array(
      'type' => 'text'
      ,'field' => 'source_text'
      ,'label' => 'Source Text'
      ,'columns' => 12
      ,'attributes' => array(
        'class' => 'large-text'
      )
    )
    ,array(
      'type' => 'text'
      ,'field' => 'source_url'
      ,'label' => 'Source URL'
      ,'columns' => 12
      ,'attributes' => array(
        'class' => 'large-text'
      ),
      'validate' => array(
        array(
          'type' => 'url'
        )
      )
    )
  )
));
